#Subrutina para mover archivos duplicados
sub mover_duplicado{
	my ($or, $name, $des) = @_;
	if (opendir (DIRH,$des)){
		@flist=readdir (DIRH);
		closedir (DIRH);
	}
	$max = 0;
	foreach(@flist){
		next if ($_ eq "." || $_ eq "..");
		my ($ndes, $path, $suffix) = fileparse($_,qr/\.[^.]*/);
		if ($ndes eq $name){
			$suffix =~ s/.//;
			if ($suffix =~ /^\d+?$/){
				if ($suffix gt $max){
					$max = $suffix;
				}
			}
		}
		if ($ndes . $suffix eq $name){
			print "nada\n";
		}
	}
	$max++;
	$new_name = $name . "." . $max;
	$new_dest = $des . $new_name;
	rename ($or, $new_dest);
}
